001  (ns halo_api.hander-test
002    (:use clojure.test)
003    (:require [halo-api.handler :refer :all]
004              [halo-api.mongo :refer :all]
005              [ring.mock.request :as mock]
006              [monger.collection :as mc]
007              [util.auth :refer :all]
008              [clojure.data.json :as json]))
009  
010  (defn user-init []
011    (def test-user (mc/insert-and-return db "user" {:gender 1
012                                               :nickname "Test User"
013                                               :city "Utopia"
014                                               :country "Utopia"
015                                               :openid "test openid"}))
016    (def test-token (save-token! (:_id test-user))))
017  
018  (defn roll-back-test []
019    (mc/remove db "token" { :user_id (:_id test-user)})
020    (mc/remove db "user" { :_id test-user}))
021  
022  (defn wrap-test [test-fn]
023    (user-init)
024    (test-fn)
025    (roll-back-test))
026  
027  (deftest test-app
028    (testing "hello world"
029      (let [response (app (mock/request :get "/hello/world"))]
030        (let [res (json/read-str (slurp (:body response)) :key-fn keyword)]
031          (is (= (:status response) 200))
032          (is (= res {:msg "Hello World!!"}))))))
033  
034  (deftest test-login
035    (testing "login"
036      (let [response (app (mock/request :get "/wechat/auth"))]
037        (is (= (:status response) 400))
038        (is (= (:body response) "Code Required!")))))
039  
040  (deftest test-orders
041    (testing "orders"
042      (let [response (app (-> (mock/request :post "/base/orders" nil)
043                              (mock/header "Authorization" test-token)
044                              (mock/body {:item "text book" :fee 4500})))]
045        (let [res (json/read-str (slurp (:body response)) :key-fn keyword)]
046          (is (= (:status response) 200))
047          (is (= res {:item "text book" :fee "4500" :status "unpaid"}))))))
048  
049  (deftest test-logout
050    (testing "logout"
051      (let [response (app (-> (mock/request :post "/base/logout" nil)
052                              (mock/header "Authorization" test-token)))]
053        (let [res (json/read-str (slurp (:body response)) :key-fn keyword)]
054          (is (= (:status response) 200))
055          (is (= res {:msg "success"}))))))
056  
057  (use-fixtures :once wrap-test)
058  (run-tests)
