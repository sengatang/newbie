001  (ns util.auth
002      (:require [halo_api.settings :as settings]
003                [crypto.random :refer [base64]]
004                [monger.collection :as mc]
005                [halo-api.mongo :as mongo]
006                [buddy.core.nonce :as nonce]
007                [buddy.core.codecs :as codecs]
008                [buddy.auth :refer [throw-unauthorized]]))
009  
010  (defn gen-token
011        []
012        (let [randomdata (nonce/random-bytes 16)]
013             (codecs/bytes->hex randomdata)))
014  
015  (defn save-token! [user-id]
016        (let [[token this-time]  [(gen-token) (.getTime (java.util.Date.))]]
017             (:key (mc/insert-and-return mongo/db "token" {:key token
018                                                           :user_id user-id
019                                                           :expires (+ this-time settings/token-expire-time)}))))
020  (defn get-or-generate-token
021        [user-id]
022        (if (not (nil? user-id))
023          (let [record (last (mc/find-maps mongo/db "token" {:user_id user-id}))]
024            (if (< (.getTime (java.util.Date.)) (:expires record 0))
025              (:key record)
026              (save-token! user-id)))))
027  
028  (defn authenticate?
029        [token]
030        (let [record (first (mc/find-maps mongo/db "token" {:key token}))]
031             (if (< (.getTime (java.util.Date.)) (:expires record 0))
032                 (:user_id record))))
033  
034  (defn authenticate-token
035        [token]
036        #_(println "authenticate-token" token)
037        (let [user_id (authenticate? token)]
038          #_(println "userid" user_id)
039          (if (nil? user_id)
040            (throw-unauthorized)
041            (mc/find-map-by-id mongo/db "user" user_id))))
